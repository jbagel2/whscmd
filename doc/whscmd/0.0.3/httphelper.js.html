/**
    Generic HTTPGet class.
    @constructor
    @param {object} options - Options for the HTTP post.
    @param {string} options.host - host to send the request to
    @param {string} options.port - port to send the request to
    @param {string} options.apikey - apikey to use for the request
    @param {function} options.logger - logger function to log output to
    @param {string} path - URL to perform the GET on.
    @param {function} formatOutput - function to process JSON output
**/
function HTTPGet(options, path, formatOutput) {
    var logger = options.logger;

    /** @property {string} Name of the command invoking the GET. */
    this.options = {
        host: options.host,
        rejectUnauthorized: false,
        port: options.port,
        method: 'GET',
        path: '/' + path + (path.indexOf('?') == -1 ? '?' : '&') + 'key=' + options.apikey
    };
    this.logger = options.logger;
    /** @method HTTPGet.formatOutput - Output formatter to process data and output accordingly
        @param {object} data - data, in JSON format, from the GET. **/
    this.formatOutput = formatOutput || function(data) {
        console.log('Listing ' + this.name);
        console.log(JSON.stringify(data));
    };
    /** Execute the get
        @method HTTPGet.execute
        @returns {boolean} true on success, false on error **/
    this.execute = function () {
        var that = this;
        if (!options.apikey) {
            logger.error('must supply an API key option (-k)'); 
            return false;
        }
        var responseData = '';
        logger.verbose('Outbound Headers => ' + JSON.stringify(that.options));
        var clientReq = https.request(that.options, function (clientRes) {
            logger.info('StatusCode => ' + clientRes.statusCode);
            logger.info('Headers => ' + JSON.stringify(clientRes.headers));

            // Restore the chunks of data to a single string.
            clientRes.on('data', function (chunk) {
                responseData += chunk;
            });

            // Process the end event.
            clientRes.on('end', function () {
                logger.info('end received');
                if (!responseData) {
                    logger.error('no data returned');
                } else { 
                    logger.verbose('raw data => ', responseData)
                    responseData = JSON.parse(responseData);

                    if (responseData.status) {
                        logger.error(responseData.message);
                    } else {
                        that.formatOutput(responseData);
                    }
                }
            });
        }).on('error', function (e) {
            logger.error('problem with https request => ' + e.message);
            return false;
        });
        clientReq.end();
        return true;
    }
}

/**
    Generic HTTPPost class.
    @constructor
    @param {object} options - Options for the HTTP post
    @param {string} options.host - host to send the request to
    @param {string} options.port - port to send the request to
    @param {string} options.apikey - apikey to use for the request
    @param {function} options.logger - logger function to log output to
    @param {string} path - URL to perform the POST on
**/
function HTTPPost(options, path) {
    var logger = options.logger;

    /** @property {object} options - options associated with the POST */
    this.options = {
        host: options.host,
        rejectUnauthorized: false,
        port: options.port,
        method: 'POST',
        path: '/' + path + (path.indexOf('?') == -1 ? '?' : '&') + 'key=' + options.apikey,
        headers: {
            'Content-Type': 'application/json',
            'Content-Length': 0,
            'Accept': 'application/json'
        }
    };
        
    /** Execute the post
        @method HTTPPost.execute
        @param {object} postData - data to post in JSON format
        @returns {boolean} true on success, false on error **/
    this.execute = function (postData) {
        postData = JSON.stringify(postData);
        var that = this;
        if (!options.apikey) {
            logger.error('must supply an API key option (-k)'); 
            return false;
        }

        that.options.headers["Content-Length"] = postData.length;

        var responseData = '';
        logger.verbose('Outbound Headers => ' + JSON.stringify(that.options));
        var clientReq = https.request(that.options, function (clientRes) {
            logger.info('Status Code => ' + clientRes.statusCode);
            logger.info('Headers => ' + JSON.stringify(clientRes.headers));

            // Restore the chunks of data to a single string.
            clientRes.on('data', function (chunk) {
                responseData += chunk;
            });

            // Process the end event.
            clientRes.on('end', function () {
                logger.info('end received');
                try {
                    responseData = JSON.parse(responseData);
                    if (responseData.message) {
                        logger.error(responseData.message);
                        return false;
                    } else { 
                        console.log('Response => ', responseData)
                    }
                } catch (e) {
                    logger.error('problem with parsing response => ' + e.message);
                    return false;                    
                }
            });
        }).on('error', function (e) {
            logger.error('problem with https request => ' + e.message);
            return false;
        });

        logger.verbose('Outbound POST Data => ' + postData);
        clientReq.write(postData);
        clientReq.end();
        return true;
    }
}
